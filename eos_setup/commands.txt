! Create the SSL profiles and bind the certs/keys
management security
   ssl profile eapi
      certificate eapi.crt key eapi.key
   !
   ssl profile gnmi
      certificate gnmi.crt key gnmi.key
   !
   ssl profile restconf
      certificate rest.crt key rest.key
   !
   ssl profile sshx
      certificate sshx.crt key sshx.key
